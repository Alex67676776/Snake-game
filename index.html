    <div id="score">Счёт: 0</div>

    <canvas id="gameCanvas"></canvas>

    <div id="gameOver">Игра окончена! Нажмите на экран для перезапуска</div>

   

    <div class="controls">

        <div class="btn empty"></div>

        <div class="btn" id="btnUp">▲</div>

        <div class="btn empty"></div>

        <div class="btn" id="btnLeft">◀</div>

        <div class="btn" id="btnRestart">↻</div>

        <div class="btn" id="btnRight">▶</div>

        <div class="btn empty"></div>

        <div class="btn" id="btnDown">▼</div>

        <div class="btn empty"></div>

    </div>

    <script>

        // Адаптация размера канваса под экран

        const canvas = document.getElementById('gameCanvas');

        const size = Math.min(window.innerWidth - 40, window.innerHeight - 300, 400);

        canvas.width = size;

        canvas.height = size;

       

        const ctx = canvas.getContext('2d');

        const scoreElement = document.getElementById('score');

        const gameOverElement = document.getElementById('gameOver');

        const gridSize = Math.floor(size / 20);

        const tileCount = 20;

        let snake = [];

        let food = {};

        let direction = { x: 0, y: 0 };

        let nextDirection = { x: 0, y: 0 };

        let score = 0;

        let gameRunning = true;

        let gameStarted = false;

        function init() {

            snake = [

                { x: 10, y: 10 }

            ];

            direction = { x: 0, y: 0 };

            nextDirection = { x: 0, y: 0 };

            score = 0;

            gameRunning = true;

            gameStarted = false;

            gameOverElement.style.display = 'none';

            updateScore();

            generateFood();

        }

        function generateFood() {

            do {

                food = {

                    x: Math.floor(Math.random() * tileCount),

                    y: Math.floor(Math.random() * tileCount)

                };

            } while (snake.some(segment => segment.x === food.x && segment.y === food.y));

        }

        function updateScore() {

            scoreElement.textContent = `Счёт: ${score}`;

        }

        function gameLoop() {

            if (!gameRunning) return;

           

            if (gameStarted) {

                update();

            }

            draw();

            setTimeout(gameLoop, 100);

        }

        function update() {

            direction = { ...nextDirection };

           

            if (direction.x === 0 && direction.y === 0) return;

            const head = { ...snake[0] };

            head.x += direction.x;

            head.y += direction.y;

            // Столкновение со стенами

            if (head.x < 0 || head.x >= tileCount || head.y < 0 || head.y >= tileCount) {

                gameOver();

                return;

            }

            // Столкновение с собой

            if (snake.some(segment => segment.x === head.x && segment.y === head.y)) {

                gameOver();

                return;

            }

            snake.unshift(head);

            // Съедена ли еда

            if (head.x === food.x && head.y === food.y) {

                score++;

                updateScore();

                generateFood();

            } else {

                snake.pop();

            }

        }

        function draw() {

            ctx.fillStyle = '#000';

            ctx.fillRect(0, 0, canvas.width, canvas.height);

            // Еда

            ctx.fillStyle = '#ff4444';

            ctx.fillRect(food.x * gridSize + 1, food.y * gridSize + 1, gridSize - 2, gridSize - 2);

            // Змейка

            snake.forEach((segment, index) => {

                ctx.fillStyle = index === 0 ? '#4CAF50' : '#388E3C';

                ctx.fillRect(segment.x * gridSize + 1, segment.y * gridSize + 1, gridSize - 2, gridSize - 2);

            });

        }

        function gameOver() {

            gameRunning = false;

            gameOverElement.style.display = 'block';

        }

        function setDirection(dx, dy) {

            // Блокировка противоположного направления

            if (direction.x === -dx && direction.y === -dy) return;

            nextDirection = { x: dx, y: dy };

            gameStarted = true;

        }

        // Кнопки управления

        document.getElementById('btnUp').addEventListener('click', () => setDirection(0, -1));

        document.getElementById('btnDown').addEventListener('click', () => setDirection(0, 1));

        document.getElementById('btnLeft').addEventListener('click', () => setDirection(-1, 0));

        document.getElementById('btnRight').addEventListener('click', () => setDirection(1, 0));

        document.getElementById('btnRestart').addEventListener('click', () => { init(); gameLoop(); });

        // Поддержка свайпов

        let touchStartX = 0;

        let touchStartY = 0;

        canvas.addEventListener('touchstart', (e) => {

            if (!gameRunning) {

                init();

                gameLoop();

                return;

            }

            touchStartX = e.touches[0].clientX;

            touchStartY = e.touches[0].clientY;

            e.preventDefault();

        });

        canvas.addEventListener('touchmove', (e) => e.preventDefault());

        canvas.addEventListener('touchend', (e) => {

            if (!gameRunning) return;

           

            const dx = e.changedTouches[0].clientX - touchStartX;

            const dy = e.changedTouches[0].clientY - touchStartY;

           

            if (Math.abs(dx) < 20 && Math.abs(dy) < 20) return; // Слишком короткий свайп

           

            if (Math.abs(dx) > Math.abs(dy)) {

                setDirection(dx > 0 ? 1 : -1, 0);

            } else {

                setDirection(0, dy > 0 ? 1 : -1);

            }

        });

        // Клик по канвасу для перезапуска

        canvas.addEventListener('click', () => {

            if (!gameRunning) {

                init();

                gameLoop();

            }

        });

        // Запуск

        init();

        gameLoop();

    </script>
